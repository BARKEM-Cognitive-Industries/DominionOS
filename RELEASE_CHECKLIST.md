```bash

cargo build --release

# Generate bootable images
cargo bootimage --release

# Create checksums
sha256sum dominionos.iso dominionos.img > SHA256SUMS

# Create tarball
tar czf dominionos-1.0-preview-release.tar.gz \
  dominionos-1.0-preview.iso \
  dominionos-1.0-preview.img \
  RELEASE_NOTES.md \
  SHA256SUMS
```

## ✅ Final Verification Before Release

Run this before hitting "Publish Release":

```bash
# 1. Clean build
cargo clean
cargo build --release 2>&1 | tee build.log
[ $? -eq 0 ] && echo "✅ Build successful" || echo "❌ Build failed"

# 2. Tests
cargo test --release --quiet 2>&1 | tee test.log
[ $? -eq 0 ] && echo "✅ Tests passed" || echo "❌ Tests failed"

# 3. Benchmarks
cargo run --release --bin bench 2>&1 | tee bench.log
[ $? -eq 0 ] && echo "✅ Benchmarks ran" || echo "❌ Benchmarks failed"

# 4. Boot test
timeout 30 ./run.ps1
[ $? -eq 0 ] && echo "✅ Boots successfully" || echo "⚠️ Boot test incomplete"

# 5. Artifacts
[ -f dominionos.iso ] && echo "✅ ISO exists" || echo "❌ ISO missing"
[ -f dominionos.img ] && echo "✅ IMG exists" || echo "❌ IMG missing"

# 6. Documentation
[ -f README_RELEASE.md ] && echo "✅ Release README exists" || echo "❌ Missing"
[ -f LICENSE.md ] && echo "✅ LICENSE exists" || echo "❌ Missing"
[ -f CONTRIBUTING.md ] && echo "✅ CONTRIBUTING exists" || echo "❌ Missing"

echo ""
echo "Release checklist complete!"
```
